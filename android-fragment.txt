2. on MyActivity.kt

        findViewById<Button>(R.id.upload).setOnClickListener {
            supportFragmentManager
                .beginTransaction()
                .replace(R.id.tablet, UploadFragment())
                .commit()
        }

        findViewById<Button>(R.id.download).setOnClickListener {
            supportFragmentManager
                .beginTransaction()
                .replace(R.id.tablet, DownloadFragment())
                .commit()
        }
        findViewById<Button>(R.id.backup).setOnClickListener {
            supportFragmentManager
                .beginTransaction()
                .replace(R.id.tablet, BackupFragment())
                .commit()
        }
