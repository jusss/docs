1. vim ~/.ctags
    -python-kinds=-i

2. cd project
   find . |ctags -R -o ~/.vim/python-tags -
    or ctags -R -o ~/.vim/python-tags ~/project1 ~/project2

3. vim ~/.vimrc
    set tags=~/.vim/python-tags

put cursor on a word, C-] will jump to its definition, C-t will jump back, 
C-w C-] will open definition in a horizontal split
